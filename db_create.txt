create table manga_name(
id serial primary key,
name varchar(255),
url varchar(255),
description text
);

create table manga_chapter (
id serial primary key,
id_manga_name integer references manga_name(id),
name_manga_chapter varchar(255)
);

create table manga_page(
id serial primary key,
id_manga_chapter integer references manga_chapter(id),
page_namber integer,
url_page varchar(255)
);
